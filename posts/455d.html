1java内部类

1.1内部类的作用

  1. 提供更好的封装；即不允许同一个包中的类访问该类
  2. 匿名内部类使用于创建只需要使用一次的类

1.2分类（静态内部类，非静态内部类）

  • 成员内部类（为类成员）
  • 局部内部类
  • 匿名内部类

1.3内部类与外部类的区别

  1. 外部类不可以使用private protect static 修饰符

    • 非private: privte类只允许自身访问类属性，修饰外部类无意义，用private修饰表示该类为内部类

    • 非protect: 该修饰符只表示拥有继承关系的父子类中，protect类可以被同一个包中和拥有父子关系的类访问，禁止用于外部类

    • 非static: static 修饰的类将类变成类相关，而不是实例相关，而外部类的上一级程序单元为包，故不可用static修饰

  2. 只有静态类中才可以拥有静态成员

    1.4静态内部类

    1. 注意：
      • 静态与非静态之间的相互访问
      • 访问静态类的静态方法 类名.静态方法名
    2. 代码实现
    package Static_IntenalClass;

 public class Static_outer {

    private  String out_static_var="外部类成员1";
    static  int out_var1=101;

    public static class Static_inner{
        private String out_var2="非静态内部类成员";
        static int inner_var=100;

        public static void inner_static_method(){
            System.out.println("内部静态方法执行");
        }
        public  void inner_method(){
            System.out.println(" 内部类非静态方法执行");
            //
            System.out.println(out_var1);
            //无法在静态类中使用外部类非静态字段 或者外部类非静态方法
            //System.out.println(out_static_var);
        }
    }

    public  void out() {
        System.out.println(Static_inner.inner_var);
        System.out.println(out_static_var);

        Static_inner inner = new Static_inner();
         //外部类中访问静态内部类的静态方法：类名.静态方法名
        Static_inner.inner_static_method();
        //外部类中访问静态内部类的非静态方法，需要创建内部类对象
        inner.inner_method();
    }

     public static void main(String[] args) {
        new Static_outer().out();


     }
}

    1.5成员内部类

    1. 注意：

      • 该内部类中不可以有静态方法，成员

      • 在外部类访问内部类。。。可参考静态内部类 跳转到页内链接

    2. package Member_InternalClass;
//成员内部类
class Outer {
    private static String name="Apple";
    private  boolean isEmpty=true;
    public  void test(){
        System.out.println("outer");
        Inner inner = new Inner();
        inner.test();
        System.out.println(inner.name);
    }

    public static void main(String[] args) {
        //在main方法中由外部类创建内部类对象
        Outer outer = new Outer();
        Inner inner1 = outer.new Inner();

        System.out.println(inner1.name);
        outer.test();
    }

    class Inner{
        //成员内部类不可以静态的方法或成员
        //该内部类可以访问外部类的全部方法或成员
        //由于内部类不存在静态的方法或成员，因此从外部类访问内部类时： 外部类对象>>内部类对象
        public  String name="HuaWei";
        public void test(){
            System.out.println("inner");
            //访问与外部类重名的变量
            System.out.println(Outer.this.name+Outer.this.isEmpty);
        }
    }
}

    1.6局部内部类

    1. 注意：定义在方法体内

      • 定义在静态方法内，只可以访问外部类的静态的成员和方法。
      • 定义在实例方法中是，可以访问外部类的全部方法和成员。

    2. 代码实现：

      package Local_InternalClass;

 class Outer {
      private String name="冬瓜";
      private static int id;

      public void setId (int i){
          this.id=id;
      }
      public int getId(){
          return this.id;
      }
      public String getName(){
          return name;
      }

     public void test1(){
          //与内部类同名的外部类方法
         getName();
         //inner inner = new inner(); 无法从外部类访问局部内部类
         System.out.println( name);

     }
     public  void  inner_test2(){
         class  inner{
             private String name="西瓜";
             // private static int id; 报错 局部内部类中不能有Static声明
             private int id;

             public inner(int id){
                 this.id=id;
             }
             public String getName(){
                 //访问外部类的静态变量
                 if(Outer.id>=100) return name;
                 //访问外部类的私有成员变量
                 else if (id<100) return Outer.this.name;
                 else return "NONO";
             }
         }
    }
}

    1.7匿名内部类、

    1. 注意:

      • 匿名内部类用于需要实现一个接口或者父类，且只使用一次。

      • 关系：每个匿名内部类对象都拥有一个外部类对象，但外部类对象并非如此因此，存在匿名内部类对象就可以访问外部类，而外部内对象却不可以显式的访问内部类

      • 格式要求：

        new 接口或者父类名（参数）{
//匿名内部类方法体中需要实现接口或者父类中的**所有**抽象方法
}

    2. 实现代码：

      package Lambda;

/**
 * 1.Device抽象类拥有2个抽象方法；2个构造函数（无参数与有参数）
 * 2.用于匿名内部类的父类*/
public  abstract class Device {
    public abstract String getName(String a);
    public abstract double getPrice(double d);
    public Device(){}
    public Device(boolean isUsed){
    }
}

package Lambda;
 class Outer {//outer是一个外部类
    //外部类的实例字段
    private static int id;
    //外部类test1（）方法
    /**
     * test1()方法：
     * @参数：Device类型的对象
     * @无返回值*/
    public void test1(Device aDevice){
        System.out.println( "产品信息："+aDevice.getName("LEGION")+"    "+aDevice.getPrice(7999.0)
                +"  "+"id:"+this.id
                );
    }

    public static void main(String[] args) {

        Outer outer = new Outer();

        /**
         * 由于test1（）需要Device类型的对象做参数，且只需要使用一次！使用内部匿名类
         * 格式：  new 抽象类名（）或接口名{}
         * 要求：该类方法体中需要实现抽象类或者接口的 “所有” 抽象方法
         * */
        outer.test1(new Device(){
            public String getName(String a){
                return a;
            }
            public double getPrice(double d){
                return d;
            }
        });
        System.out.println("-----------------------");

        new Device(true){
              int year=2020;//匿名内部类的局部变量，自从java8开始，内部类中的方法允许访问                               非final局部变量
                             //匿名内部类中不可以有静态的成员和方法

            @Override
            public  String getName(String a) {
                return null;
            }

            @Override
            public double getPrice(double d) {
                return 0;
            }
            public void test2(){
                System.out.println("END");
                System.out.println(year);
            }
            //内部类的初始化块
            {
                id=1;
            }

        //调用内部类的方法
        }.test2();
    }
}
